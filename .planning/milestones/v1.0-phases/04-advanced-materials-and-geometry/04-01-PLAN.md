```python
def sample_angular_distribution(n, normal, theta_deg, intensity, rng) -> np.ndarray:
    # Builds 2048-point interpolation grid, sin(theta) weighted CDF, inversion
    grid = np.linspace(theta_rad[0], theta_rad[-1], 2048)
    interp_i = np.interp(grid, theta_rad, intensity, left=0.0, right=0.0)
    weights = interp_i * np.sin(grid)
    cdf = np.cumsum(weights) / np.cumsum(weights)[-1]
    sample_theta = np.interp(rng.uniform(0, 1, n), cdf, grid)
```
