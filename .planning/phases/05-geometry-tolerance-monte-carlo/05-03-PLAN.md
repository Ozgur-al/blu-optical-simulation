```python
"""Ensemble tolerance Monte Carlo dialog.

Runs N realizations of the current project sampled from tolerance distributions and
displays KPI distributions (histogram, P5/P50/P95) and sensitivity ranking.
"""
from __future__ import annotations

import copy

import numpy as np
import pyqtgraph as pg
from PySide6.QtCore import QThread, Signal, Qt
from PySide6.QtWidgets import (
    QDialog, QVBoxLayout, QHBoxLayout, QFormLayout, QSplitter,
    QComboBox, QDoubleSpinBox, QSpinBox, QLabel, QWidget,
    QPushButton, QProgressBar, QTableWidget, QTableWidgetItem,
    QHeaderView, QMessageBox, QCheckBox, QGroupBox,
)

from backlight_sim.core.project_model import Project
from backlight_sim.core.kpi import compute_scalar_kpis


# ---------------------------------------------------------------------------
# Background QThread
# ---------------------------------------------------------------------------

class _EnsembleThread(QThread):
    """Runs N ensemble realizations in a background thread.

    Emits step_done(idx, result, project_clone) after each member.
    Emits sweep_finished() when all members complete or cancel is called.
    """
    step_done = Signal(int, object, object)   # (member_idx, SimulationResult, Project clone)
    sweep_finished = Signal()

    def __init__(
        self,
        base_project: Project,
        n_members: int,
        mode: str,      # "oat" | "sobol"
        seed: int,
    ) -> None:
        super().__init__()
        self._base = base_project
        self._n = min(max(1, n_members), 500)   # DoS clamp [1, 500]
        self._mode = mode
        self._seed = seed & 0x7FFFFFFF          # int32 mask (Phase 4 pattern)
        self._cancelled = False
        self._member_projects: list[Project] = []   # for worst-case drill-down

    def cancel(self) -> None:
        self._cancelled = True

    def run(self) -> None:
        from backlight_sim.sim.ensemble import build_oat_sample, build_sobol_sample
        from backlight_sim.sim.tracer import RayTracer

        if self._mode == "sobol":
            samples = build_sobol_sample(self._base, self._n, self._seed)
        else:
            samples = build_oat_sample(self._base, self._seed)

        for i, (proj, _param_info) in enumerate(samples):
            if self._cancelled:
                break
            try:
                result = RayTracer(proj).run()
            except Exception:
                continue    # skip failed members; do not crash the thread
            self._member_projects.append(proj)
            self.step_done.emit(i, result, proj)

        self.sweep_finished.emit()
```

    # ------------------------------------------------------------------
    # Run / cancel
    # ------------------------------------------------------------------

    def _run_ensemble(self) -> None:
        if not [s for s in self._project.sources if s.enabled]:
            QMessageBox.warning(self, "No Active Sources",
                                "Enable at least one source before running.")
            return
        if not self._project.detectors and not self._project.sphere_detectors:
            QMessageBox.warning(self, "No Detectors",
                                "Add a detector before running.")
            return

        n = self._n_spin.value()
        mode = "sobol" if self._sobol_check.isChecked() else "oat"

        self._all_kpis.clear()

    # ------------------------------------------------------------------
    # Step handlers
    # ------------------------------------------------------------------

    def _on_step_done(self, idx: int, result, proj: Project) -> None:
        self._progress.setValue(idx + 1)
        self._member_projects.append(proj)
        kpis = compute_scalar_kpis(result)
        self._all_kpis.append(kpis)

        # Accumulate selected KPI
        selected_key = _KPI_LABEL_TO_KEY[self._kpi_combo.currentText()]
        val = kpis.get(selected_key, 0.0)
        self._kpi_values.append(val)

        self._update_histogram(self._kpi_values)
        self._update_worst_case(selected_key)

    def _on_sweep_finished(self) -> None:
        self._run_btn.setEnabled(True)
        self._cancel_btn.setEnabled(False)
        if self._thread:
            self._member_projects = self._thread._member_projects
        if len(self._all_kpis) > 1:
            self._update_sensitivity_table()
        if self._worst_project is not None:
            self._load_worst_btn.setEnabled(True)

    def _on_kpi_changed(self) -> None:
        """Re-render histogram when user switches KPI selector."""
        if not self._all_kpis:
            return
        selected_key = _KPI_LABEL_TO_KEY[self._kpi_combo.currentText()]
        self._kpi_values = [k.get(selected_key, 0.0) for k in self._all_kpis]
        self._update_histogram(self._kpi_values)
        self._update_worst_case(selected_key)

    # ------------------------------------------------------------------
    # Histogram
    # ------------------------------------------------------------------

    def _update_histogram(self, kpi_values: list[float]) -> None:
        """Live-update BarGraphItem with dynamic bin edges (pitfall 6 avoidance)."""
        if len(kpi_values) < 2:
            return
        counts, edges = np.histogram(kpi_values, bins=min(20, len(kpi_values)))
        centers = (edges[:-1] + edges[1:]) / 2
        width = float(edges[1] - edges[0]) if len(edges) > 1 else 0.05
        self._hist_item.setOpts(x=centers, height=counts.astype(float), width=width)
        if len(kpi_values) >= 5:
            p5, p50, p95 = np.percentile(kpi_values, [5, 50, 95])
            self._p5_line.setValue(p5)
            self._p50_line.setValue(p50)
            self._p95_line.setValue(p95)
            self._percentile_label.setText(
                f"P5: {p5:.4f} | P50: {p50:.4f} | P95: {p95:.4f}"
            )

    # ------------------------------------------------------------------
    # Sensitivity table
    # ------------------------------------------------------------------

    def _update_sensitivity_table(self) -> None:
        """Populate sensitivity table using OAT indices from accumulated KPIs."""
        from backlight_sim.sim.ensemble import (
            compute_oat_sensitivity, _active_tolerance_params
        )
        if len(self._all_kpis) < 2:
            return
        params = _active_tolerance_params(self._project)
        if not params:
            return

        baseline_kpis = self._all_kpis[0]    # index 0 = OAT baseline
        perturbed_kpis = self._all_kpis[1:]  # indices 1..k = perturbed
        param_names = [p[0] for p in params]
        param_sigmas = [p[1] for p in params]

        # Trim perturbed_kpis to number of actual params (OAT: exactly k entries after baseline)
        perturbed_kpis = perturbed_kpis[:len(param_names)]
        if len(perturbed_kpis) < len(param_names):
            return   # not enough data yet

        selected_key = _KPI_LABEL_TO_KEY[self._kpi_combo.currentText()]
        sensitivity = compute_oat_sensitivity(
            baseline_kpis, perturbed_kpis, param_names, param_sigmas
        )

        sens_for_kpi = sensitivity.get(selected_key, [0.0] * len(param_names))
        rows = sorted(zip(param_names, sens_for_kpi), key=lambda x: -x[1])

        self._sens_table.setRowCount(len(rows))
        for row_idx, (pname, sens_val) in enumerate(rows):
            self._sens_table.setItem(row_idx, 0, QTableWidgetItem(pname))
            self._sens_table.setItem(row_idx, 1, QTableWidgetItem(f"{sens_val:.4f}"))

    # ------------------------------------------------------------------
    # Worst-case drill-down
    # ------------------------------------------------------------------

    def _update_worst_case(self, selected_key: str) -> None:
        """Track the member project with the worst (lowest) selected KPI."""
        if not self._all_kpis:
            return
        worst_idx = int(np.argmin([k.get(selected_key, float("inf"))
                                   for k in self._all_kpis]))
        if worst_idx < len(self._member_projects):
            self._worst_project = self._member_projects[worst_idx]
            worst_val = self._all_kpis[worst_idx].get(selected_key, 0.0)
            self._load_worst_btn.setText(
                f"Load worst case as variant ({selected_key}={worst_val:.4f})"
            )

    def _load_worst_case(self) -> None:
        if self._worst_project is None:
            return
        selected_key = _KPI_LABEL_TO_KEY[self._kpi_combo.currentText()]
        worst_val = min(k.get(selected_key, float("inf")) for k in self._all_kpis)
        name = f"Ensemble worst ({selected_key}={worst_val:.4f})"
        self.save_variant.emit(name, self._worst_project)

    # ------------------------------------------------------------------
    # Window close
    # ------------------------------------------------------------------

    def closeEvent(self, event) -> None:
        if self._thread and self._thread.isRunning():
            self._thread.cancel()
            self._thread.wait(2000)
        super().closeEvent(event)
