```python
def make_fresnel_scene():
    def _build(theta_deg: float = 0.0, rays: int = 200_000,
               n_glass: float = 1.5) -> Project:
        project = _base_project(f"fresnel_theta{theta_deg}", rays, max_bounces=4)
        # Materials
        project.materials["glass"] = Material(
            name="glass", surface_type="reflector",
            reflectance=0.0, absorption=0.0, is_diffuse=False,
            refractive_index=n_glass,
        )
        project.materials["absorber"] = Material(
            name="absorber", surface_type="absorber",
            reflectance=0.0, absorption=1.0,
        )
        # Glass slab — bottom face overridden to absorber so the second
        # interface never contributes to measured T (Pitfall 1 mitigation).
        project.solid_boxes.append(SolidBox(
            name="slab",
            center=np.array([0.0, 0.0, 0.0]),
            dimensions=(50.0, 50.0, 2.0),
            material_name="glass",
            face_optics={"bottom": "absorber"},  # verify exact key in solid_body.py
        ))
        # Source tilted about x-axis by theta_deg so incidence on top face = theta
        theta = np.radians(theta_deg)
        src_pos = np.array([0.0, -10.0 * np.sin(theta), 5.0 + 10.0 * np.cos(theta)])
        src_dir = np.array([0.0, np.sin(theta), -np.cos(theta)])
        src_dir = src_dir / np.linalg.norm(src_dir)
        # Narrow pencil beam — custom angular distribution
        project.angular_distributions["pencil"] = {
            "theta_deg": [0.0, 1.0],
            "intensity": [1.0, 0.0],
        }
        project.sources.append(PointSource(
            name="src", position=src_pos,
            flux=1000.0,
            direction=src_dir,
            distribution="pencil",
            flux_tolerance=0.0,
        ))
        # REFLECTED detector above the box (mirror-reflected direction)
        # Reflected ray direction: src_dir reflected off top face (normal +z):
        #   r = d - 2(d·n)n = src_dir - 2*src_dir_z*(0,0,1)
        #     = (0, sin θ, +cos θ)
        # Place detector along that ray, above the box, far enough to not
        # block the incoming source geometry:
        refl_dir = np.array([0.0, np.sin(theta), np.cos(theta)])
        det_pos = np.array([0.0, 10.0 * np.sin(theta), 1.0 + 10.0 * np.cos(theta)])
        # Detector plane perpendicular to refl_dir, u_axis=x, v_axis=refl_dir × x
        u_axis = np.array([1.0, 0.0, 0.0])
        v_axis = np.cross(refl_dir, u_axis)
        v_axis = v_axis / np.linalg.norm(v_axis)
        project.detectors.append(DetectorSurface(
            name="reflected",
            center=det_pos,
            u_axis=u_axis,
            v_axis=v_axis,
            size=(100.0, 100.0),
            resolution=(10, 10),
        ))
```

```python
"""GOLD-03: Fresnel transmittance at air→glass interface, parametrized over 5 angles.

Routed to Python path (SolidBox — tracer.py:264). Uses face_optics={'bottom':'absorber'}
on the SolidBox to KILL the exit-face second interface (Pitfall 1 mitigation per
RESEARCH §Case 3); measurement is via reflected-flux detector, T = 1 - R_measured.

200k rays/angle, ±0.02 absolute tolerance per RESEARCH §Tolerance table (6–13× MC stderr).
"""
from __future__ import annotations

import numpy as np
import pytest

from backlight_sim.sim.tracer import RayTracer
from backlight_sim.golden.cases import GoldenResult
from backlight_sim.tests.golden.references import fresnel_transmittance_unpolarized


@pytest.mark.parametrize("theta_deg", [0, 30, 45, 60, 80])
def test_fresnel_transmittance_matches_analytic(
    theta_deg, make_fresnel_scene, assert_within_tolerance,
):
    rays = 200_000
    project = make_fresnel_scene(theta_deg=theta_deg, rays=rays, n_glass=1.5)

    # Confirm pitfall-1 guard is wired: the SolidBox has the absorber override.
    slab = project.solid_boxes[0]
    assert getattr(slab, "face_optics", None) is not None, (
        "Pitfall 1 guard missing: SolidBox must have face_optics override on "
        "one face to kill the second interface"
    )
    face_optics = slab.face_optics or {}
    assert "absorber" in face_optics.values(), (
        "Pitfall 1 guard missing: face_optics must map one face to 'absorber' — "
        "otherwise measured T = T(θ_i)·T(θ_t), not T(θ_i)"
    )

    result = RayTracer(project).run()
    reflected_flux = float(result.detectors["reflected"].total_flux)
    source_flux = float(project.sources[0].flux)
    R_measured = reflected_flux / source_flux
    T_measured = 1.0 - R_measured
    T_expected = fresnel_transmittance_unpolarized(np.radians(theta_deg), 1.0, 1.5)

    assert_within_tolerance(GoldenResult(
        name=f"fresnel_T_theta={theta_deg}",
        expected=T_expected,
        measured=T_measured,
        residual=abs(T_measured - T_expected),
        tolerance=0.02,
        rays=rays,
        notes=f"air→glass n=1.5, reflected-flux topology, single interface",
    ))
```

```python
"""GOLD-05: Prism dispersion vs Snell's law at 3 wavelengths.

CLOSES the `project_spectral_ri_testing.md` memory flag: the solid-body spectral
n(λ) refraction path was smoke-tested but never physically verified. This test
does both (a) per-λ Snell-angle accuracy AND (b) the dispersion-detection guard
Δθ(450→650) > 0.1° (Pitfall 3) — without (b), a silent fallback to scalar
refractive_index would produce zero dispersion and pass an ambiguous tolerance.

Routed to Python path (SolidPrism + spd='mono_*' — tracer.py:631, :1495).
500k rays/λ, ±0.25° per wavelength, ±25% relative on dispersion.

See RESEARCH.md §Case 5 + §Pitfall 3.
"""
from __future__ import annotations

import numpy as np
import pytest

from backlight_sim.sim.tracer import RayTracer, _project_uses_cpp_unsupported_features
from backlight_sim.golden.cases import GoldenResult
from backlight_sim.tests.golden.references import snell_exit_angle


_APEX_DEG = 60.0   # n_sides=3 equilateral prism per conftest builder
_THETA_IN_DEG = 20.0


def _peak_theta_rad(candela_grid: np.ndarray) -> float:
    """Peak polar-bin angle (radians) of a far-field candela grid."""
    peak_idx = np.unravel_index(np.argmax(candela_grid), candela_grid.shape)
    n_theta = candela_grid.shape[0]
    return float((peak_idx[0] + 0.5) * np.pi / n_theta)


@pytest.mark.parametrize("wavelength_nm,n_expected", [
    (450, 1.5252),
    (550, 1.5187),
    (650, 1.5145),
])
def test_prism_exit_angle_matches_snell(
    wavelength_nm, n_expected, make_prism_scene, assert_within_tolerance,
):
    rays = 500_000
    project = make_prism_scene(
        wavelength_nm=wavelength_nm,
        apex_deg=_APEX_DEG,
        theta_in_deg=_THETA_IN_DEG,
        rays=rays,
    )
    # Pitfall-3 preconditions: without ALL THREE, dispersion collapses silently.
    assert project.sources[0].spd == f"mono_{wavelength_nm}", (
        "Pitfall 3 guard: source.spd must be f'mono_{wavelength_nm}' "
        "to trigger has_spectral gate (tracer.py:631)"
    )
    assert "bk7" in project.spectral_material_data, (
        "Pitfall 3 guard: spectral_material_data['bk7'] missing — dispatch "
        "at tracer.py:1495 will silently fall back to scalar n"
    )
    assert "refractive_index" in project.spectral_material_data["bk7"], (
        "Pitfall 3 guard: spectral_material_data['bk7']['refractive_index'] missing"
    )
    # Must route to Python (solid prism + spectral SPD)
    assert _project_uses_cpp_unsupported_features(project), (
        "Prism scene must route to Python path (SolidPrism + mono_* spd)"
    )

    result = RayTracer(project).run()
    sd = result.sphere_detectors["farfield"]
    assert sd.candela_grid is not None
    theta_measured = _peak_theta_rad(sd.candela_grid)
    theta_expected = snell_exit_angle(
        np.radians(_THETA_IN_DEG), n_expected, np.radians(_APEX_DEG),
    )
    assert not np.isnan(theta_expected), (
        f"Reference Snell predicts TIR at λ={wavelength_nm} — "
        f"geometry (apex={_APEX_DEG}°, θ_in={_THETA_IN_DEG}°, n={n_expected}) unusable"
    )
    residual_deg = float(abs(theta_measured - theta_expected) * 180.0 / np.pi)

    assert_within_tolerance(GoldenResult(
        name=f"prism_theta_lambda={wavelength_nm}",
        expected=float(theta_expected * 180.0 / np.pi),
        measured=float(theta_measured * 180.0 / np.pi),
        residual=residual_deg,
        tolerance=0.25,
        rays=rays,
        notes=f"λ={wavelength_nm} nm, n={n_expected}, apex={_APEX_DEG}°, θ_in={_THETA_IN_DEG}°",
    ))


def test_prism_dispersion_is_nonzero(make_prism_scene, assert_within_tolerance):
    """MEMORY-FLAG CLOSURE: Δθ(450 → 650) MUST be > 0.1°.

    This is the regression guard from RESEARCH §Pitfall 3. If the
    spectral_material_data dispatch at tracer.py:1495 silently falls back
    to scalar `refractive_index` (key mismatch, missing sub-key, spd starts
    with 'mono' but has_spectral check is broken, etc.), all three wavelengths
    will refract identically and Δθ → 0. A simple per-λ tolerance check can
    still PASS with Δθ=0 because all three expected angles land within 0.25°
    of a single bin. This test explicitly rejects that failure mode.

    Closes: ~/.claude/projects/.../memory/project_spectral_ri_testing.md
    """
    rays = 500_000
    thetas_deg = {}
    for wavelength_nm in (450, 650):
        project = make_prism_scene(
            wavelength_nm=wavelength_nm,
            apex_deg=_APEX_DEG,
            theta_in_deg=_THETA_IN_DEG,
            rays=rays,
        )
        result = RayTracer(project).run()
        sd = result.sphere_detectors["farfield"]
        thetas_deg[wavelength_nm] = _peak_theta_rad(sd.candela_grid) * 180.0 / np.pi

    dispersion_deg = abs(thetas_deg[450] - thetas_deg[650])
    # Per RESEARCH §Case 5: predicted Δθ(450→650) ≈ 0.4° for BK7 at apex=60°, θ_in=20°
    # Pass criterion: dispersion must be GREATER than 0.1° (much more than zero).
    assert dispersion_deg > 0.1, (
        f"Dispersion test FAILED — Δθ(450→650) = {dispersion_deg:.4f}° < 0.1°. "
        f"This indicates the spectral_material_data dispatch at tracer.py:1495 "
        f"is NOT being exercised — all wavelengths are refracting with the "
        f"scalar refractive_index fallback. The memory flag "
        f"`project_spectral_ri_testing.md` is NOT closed by this test run."
    )

    # Also pack a GoldenResult for the CLI report
    # Expected Δθ ≈ 0.4° for BK7 apex=60° θ_in=20° — allow ±25% relative
    # (MC stderr + analytical approximation tolerance)
    expected_dispersion_deg = abs(

      - snell_exit_angle(np.radians(_THETA_IN_DEG), 1.5145, np.radians(_APEX_DEG))
    ) * 180.0 / np.pi
    rel_residual = abs(dispersion_deg - expected_dispersion_deg) / max(
        expected_dispersion_deg, 1e-6,
    )
    assert_within_tolerance(GoldenResult(
        name="prism_dispersion_guard",
        expected=expected_dispersion_deg,
        measured=dispersion_deg,
        residual=rel_residual,
        tolerance=0.25,
        rays=rays,
        notes="MEMORY FLAG CLOSURE — project_spectral_ri_testing.md",
    ))
```
