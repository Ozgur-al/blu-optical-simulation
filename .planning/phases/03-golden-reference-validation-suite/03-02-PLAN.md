**Step C — Fill in `make_specular_mirror_scene` in conftest.py (per RESEARCH §Case 4 + Pitfall 2):**

The mirror is a tilted Rectangle. Build it rotated about the x-axis by angle `theta_deg` so the normal is `(0, sin(θ), cos(θ))`. Source at position `(0, h*sin(2θ), h*cos(2θ))` for some height h, aimed at origin — so the incidence angle relative to the mirror normal is θ. Use `Rectangle` with explicit `u_axis`/`v_axis` rather than `axis_aligned()`.

```python
@pytest.fixture
def make_specular_mirror_scene():
    def _build(theta_deg: float = 30.0, rays: int = 100_000,
               use_farfield: bool = False) -> Project:
        project = _base_project(
            f"specular_theta{theta_deg}_ff{use_farfield}", rays, max_bounces=2,
        )
        project.materials["mirror"] = Material(
            name="mirror", surface_type="reflector",
            reflectance=1.0, absorption=0.0, is_diffuse=False,
        )
        theta = np.radians(theta_deg)
        # Mirror in XY plane rotated about x-axis by theta: normal = (0, sin θ, cos θ)
        u_axis = np.array([1.0, 0.0, 0.0])
        v_axis = np.array([0.0, np.cos(theta), -np.sin(theta)])
        project.surfaces.append(Rectangle(
            name="mirror",
            center=np.array([0.0, 0.0, 0.0]),
            u_axis=u_axis,
            v_axis=v_axis,
            size=(50.0, 50.0),
            material_name="mirror",
        ))
        # Source at (0, -H*sin θ, H*cos θ), aimed at origin — incidence = theta
        H = 20.0
        src_pos = np.array([0.0, -H * np.sin(theta), H * np.cos(theta)])
        src_dir = -src_pos / np.linalg.norm(src_pos)
        project.sources.append(PointSource(
            name="src", position=src_pos,
            flux=1000.0,
            direction=src_dir,
            distribution="isotropic",  # narrow aim — most flux hits mirror
            flux_tolerance=0.0,
        ))
        if use_farfield:
            project.sphere_detectors.append(SphereDetector(
                name="farfield",
                center=np.array([0.0, 0.0, 0.0]),
                radius=1000.0,
                resolution=(360, 180),   # 0.5°/polar bin (RESEARCH §Case 4)
                mode="far_field",
            ))
        else:
            # Planar detector on predicted reflection path.
            # Reflected ray from origin points toward (0, +H sin θ, H cos θ) (mirror symmetric).
            det_pos = np.array([0.0, H * np.sin(theta), H * np.cos(theta)])
            # Detector plane perpendicular to reflected-ray direction; size large enough
            # to catch the cone from the isotropic source.
            refl_dir = det_pos / np.linalg.norm(det_pos)
            # Use Rectangle-style axis assignment — downward-facing at det_pos
            project.detectors.append(DetectorSurface(
                name="planar",
                center=det_pos,
                u_axis=np.array([1.0, 0.0, 0.0]),
                v_axis=np.cross(refl_dir, np.array([1.0, 0.0, 0.0])),
                size=(100.0, 100.0),
                resolution=(50, 50),
            ))
        return project
    return _build
```

"""GOLD-01: integrating-cavity uniformity vs finite-bounce analytical formula.

Routed to Python path (SolidBox present — tracer.py:264 predicate).
ρ = 0.9, 500k rays, 50 max_bounces, tolerance = ±2% relative per RESEARCH §Case 1.
"""
from __future__ import annotations

import numpy as np

from backlight_sim.sim.tracer import RayTracer
from backlight_sim.golden.cases import GoldenResult
from backlight_sim.tests.golden.references import integrating_cavity_irradiance


def test_integrating_cavity_uniformity(make_integrating_cavity_scene,
                                       assert_within_tolerance):
    rho = 0.9
    rays = 500_000
    radius = 50.0
    project = make_integrating_cavity_scene(radius=radius, rho=rho, rays=rays)

    result = RayTracer(project).run()
    det = result.detectors["patch"]
    assert det.total_hits > 0, "Patch detector saw zero hits — check geometry orientation"

    patch_area = float(np.prod(project.detectors[0].size))  # 10 * 10 = 100 mm²
    E_measured = float(det.total_flux) / patch_area

    # Cube cavity inner area = 6 faces × (2r)² = 24 r²
    total_area = 24.0 * radius ** 2
    E_expected = integrating_cavity_irradiance(
        phi=project.sources[0].flux,
        area=total_area,
        rho=rho,
        n_bounces=project.settings.max_bounces,
    )
    rel_residual = abs(E_measured - E_expected) / E_expected

    assert_within_tolerance(GoldenResult(

"""GOLD-02: Lambertian emitter angular intensity matches I₀·cos(θ).

Routed to Python path (SphereDetector(mode='far_field') — tracer.py:272).
500k rays, 36×72 far-field bins, θ mask ≤ 80° to avoid grazing, RMS tolerance = 0.03.
"""
from __future__ import annotations

import numpy as np

from backlight_sim.sim.tracer import RayTracer
from backlight_sim.golden.cases import GoldenResult


def test_lambertian_emitter_matches_cosine(make_lambertian_emitter_scene,
                                           assert_within_tolerance):
    rays = 500_000
    project = make_lambertian_emitter_scene(rays=rays)

    result = RayTracer(project).run()
    sd = result.sphere_detectors["farfield"]
    assert sd.candela_grid is not None, (
        "Far-field candela_grid not populated — check compute_farfield_candela "
        "ran (tracer.py:2934) and detector mode='far_field'"
    )

    # Mean over azimuth bins at each polar bin
    profile = sd.candela_grid.mean(axis=1)  # (n_theta,)
    n_theta = profile.shape[0]
    theta_centers = (np.arange(n_theta) + 0.5) * np.pi / n_theta

    # Exclude grazing θ > 80° (RESEARCH §Case 2 — cos→0 noise blowup)
    mask = theta_centers <= np.radians(80.0)
    # Also drop θ > π/2 (back hemisphere — Lambertian emits zero there)
    mask &= theta_centers <= np.pi / 2.0

    measured_masked = profile[mask]
    if measured_masked.max() <= 0:
        raise AssertionError(
            "Lambertian profile is zero everywhere — check source direction=(0,0,1) "
            "and that max_bounces=0 does not short-circuit emission"
        )

    measured_norm = measured_masked / measured_masked.max()
    expected = np.cos(theta_centers[mask])
    # Normalize expected the same way (peak = 1.0 at θ=0 already)
    rms = float(np.sqrt(np.mean((measured_norm - expected) ** 2)))

    assert_within_tolerance(GoldenResult(
        name="lambertian_cosine",
        expected=0.0,              # RMS deviation from cos(θ); target is 0
        measured=rms,
        residual=rms,
        tolerance=0.03,            # RESEARCH §Tolerance table
        rays=rays,
        notes=f"n_theta={n_theta}, mask<=80°",
    ))

"""GOLD-04: law of reflection θ_out = θ_in, dual C++/Python sub-cases.

- test_specular_angle_python_farfield: far-field SphereDetector forces Python
  (tracer.py:272) — direct angular readout.
- test_specular_angle_cpp_planar: planar DetectorSurface + no unsupported features
  → C++ path (STATE.md Wave 3). Sub-case asserts path routing explicitly to
  catch Pitfall 2 (silent fallback to Python).

100k rays each. Mirror reflectance=1.0 (no absorption). θ_i = 30°, tolerance 0.5°
(far-field) / 1.0° (planar — less angularly precise).
"""
from __future__ import annotations

import numpy as np

from backlight_sim.sim.tracer import RayTracer, _project_uses_cpp_unsupported_features
from backlight_sim.golden.cases import GoldenResult


def test_specular_angle_python_farfield(make_specular_mirror_scene,
                                        assert_within_tolerance):
    theta_deg = 30.0
    rays = 100_000
    project = make_specular_mirror_scene(
        theta_deg=theta_deg, rays=rays, use_farfield=True,
    )
    # Far-field forces Python path per tracer.py:272 — confirm
    assert _project_uses_cpp_unsupported_features(project), (
        "Far-field SphereDetector must force Python path — check scene builder"
    )

    result = RayTracer(project).run()
    sd = result.sphere_detectors["farfield"]
    assert sd.candela_grid is not None, "candela_grid missing — far-field post-processing failed"

    peak_idx = np.unravel_index(np.argmax(sd.candela_grid), sd.candela_grid.shape)
    n_theta = sd.candela_grid.shape[0]
    theta_measured = (peak_idx[0] + 0.5) * np.pi / n_theta
    theta_expected = np.radians(theta_deg)
    residual_deg = float(abs(theta_measured - theta_expected) * 180.0 / np.pi)

    assert_within_tolerance(GoldenResult(
        name="specular_reflection_python",
        expected=theta_deg,
        measured=float(theta_measured * 180.0 / np.pi),
        residual=residual_deg,
        tolerance=0.5,
        rays=rays,
        notes="far-field SphereDetector, Python path",
    ))


def test_specular_angle_cpp_planar(make_specular_mirror_scene,
                                   assert_within_tolerance):
    theta_deg = 30.0
    rays = 100_000
    project = make_specular_mirror_scene(
        theta_deg=theta_deg, rays=rays, use_farfield=False,
    )
    # MUST route to C++ per dispatch predicate — Pitfall 2 guard
    assert not _project_uses_cpp_unsupported_features(project), (
        "Planar detector scene unexpectedly routes to Python — check that "
        "no SolidBox/far-field/SPD was introduced"
    )

    result = RayTracer(project).run()
    det = result.detectors["planar"]
    assert det.total_hits > 0, "Planar detector saw zero hits — check geometry/aim"

    grid = det.grid
    total = grid.sum()
    assert total > 0
    # Centroid (u,v) in detector local coords
    ny, nx = grid.shape
    us = (np.arange(nx) + 0.5) / nx - 0.5   # normalized centered
    vs = (np.arange(ny) + 0.5) / ny - 0.5
    U, V = np.meshgrid(us, vs)
    u_cent = float((grid * U).sum() / total)
    v_cent = float((grid * V).sum() / total)
    # For correctly aimed geometry (source reflected through mirror), centroid
    # should land near (0,0) — any residual offset maps to angular residual:
    det_size = project.detectors[0].size  # (u_size, v_size)
    offset_mm = float(np.hypot(u_cent * det_size[0], v_cent * det_size[1]))
    # Geometry: planar detector is 20 mm from mirror center (from builder H=20).
    # Convert lateral offset to angular residual in degrees.
    distance = 20.0
    residual_deg = float(np.degrees(np.arctan2(offset_mm, distance)))

    assert_within_tolerance(GoldenResult(
        name="specular_reflection_cpp",
        expected=theta_deg,            # reporting only — residual is lateral
        measured=theta_deg + residual_deg,
        residual=residual_deg,
        tolerance=1.0,                 # planar detector is less precise than far-field
        rays=rays,
        notes="planar DetectorSurface, C++ path",
    ))
