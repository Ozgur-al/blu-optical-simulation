```python
Both writers are side-effect-only — they return None and write to `path`.

def _try_import_matplotlib():
    try:
        import matplotlib
        matplotlib.use("Agg")
        import matplotlib.pyplot as plt
        return plt
    except ImportError:
        return None


def _fresnel_plot_base64(results: list[GoldenResult]) -> str:
    """Fresnel T_measured vs T_analytical scatter. Returns base64 PNG or ''."""
    fresnel_results = [r for r in results if r.name.startswith("fresnel_T_theta=")]
    if len(fresnel_results) < 2:
        return ""
    plt = _try_import_matplotlib()
    if plt is None:
        return ""
    thetas = []
    measured = []
    expected = []
    for r in fresnel_results:
        # Name format: fresnel_T_theta=30
        theta = float(r.name.split("=")[-1])
        thetas.append(theta)
        measured.append(r.measured)
        expected.append(r.expected)
    idx = np.argsort(thetas)
    thetas = np.array(thetas)[idx]
    measured = np.array(measured)[idx]
    expected = np.array(expected)[idx]
    fig, ax = plt.subplots(figsize=(6, 4))
    theta_fine = np.linspace(0, 90, 200)
    # Analytical curve — for reference. Use local import of references for math.
    from backlight_sim.tests.golden.references import fresnel_transmittance_unpolarized
    T_curve = np.array([
        fresnel_transmittance_unpolarized(np.radians(t), 1.0, 1.5)
        for t in theta_fine
    ])
    ax.plot(theta_fine, T_curve, "b-", label="Analytical (n=1.5)", linewidth=1.5)
    ax.plot(thetas, measured, "ro", label="Measured", markersize=7)
    ax.set_xlabel("Incidence angle θ (°)")
    ax.set_ylabel("Transmittance T(θ)")
    ax.set_title("Fresnel T(θ) — golden reference")
    ax.grid(True, alpha=0.3)
    ax.legend()
    buf = _io.BytesIO()
    fig.savefig(buf, format="png", dpi=100, bbox_inches="tight")
    plt.close(fig)
    buf.seek(0)
    return base64.b64encode(buf.read()).decode("ascii")


def _prism_plot_base64(results: list[GoldenResult]) -> str:
    """Prism θ_exit vs λ scatter. Returns base64 PNG or ''."""
    prism_results = [r for r in results if r.name.startswith("prism_theta_lambda=")]
    if len(prism_results) < 2:
        return ""
    plt = _try_import_matplotlib()
    if plt is None:
        return ""
    lambdas = []
    measured = []
    expected = []
    for r in prism_results:
        lam = float(r.name.split("=")[-1])
        lambdas.append(lam)
        measured.append(r.measured)
        expected.append(r.expected)
    idx = np.argsort(lambdas)
    lambdas = np.array(lambdas)[idx]
    measured = np.array(measured)[idx]
    expected = np.array(expected)[idx]
    fig, ax = plt.subplots(figsize=(6, 4))
    ax.plot(lambdas, expected, "b-o", label="Snell expected", linewidth=1.5, markersize=7)
    ax.plot(lambdas, measured, "r^", label="Measured", markersize=9)
    ax.set_xlabel("Wavelength λ (nm)")
    ax.set_ylabel("Exit angle θ (°)")
    ax.set_title("Prism dispersion — golden reference")
    ax.grid(True, alpha=0.3)
    ax.legend()
    buf = _io.BytesIO()
    fig.savefig(buf, format="png", dpi=100, bbox_inches="tight")
    plt.close(fig)
    buf.seek(0)
    return base64.b64encode(buf.read()).decode("ascii")

def write_markdown_report(results: Iterable[GoldenResult], path: Path) -> None:
    """Text-only markdown summary. Works without matplotlib.

    Produces a header + one-row-per-case table. Every result.name appears
    so the integration test can verify presence by substring match.
    """
    results_list = list(results)
    total = len(results_list)
    passed = sum(1 for r in results_list if r.passed)
    lines: list[str] = [
        "# Golden-Reference Validation Report",
        "",
        f"**Generated:** {datetime.now().isoformat(timespec='seconds')}",
        f"**Python:** {platform.python_version()}",
        f"**Overall:** {passed}/{total} cases passed",
        "",
        "| Case | Expected | Measured | Residual | Tolerance | Rays | Status |",
        "|------|----------|----------|----------|-----------|------|--------|",
    ]
    for r in results_list:
        status = "PASS" if r.passed else "FAIL"
        lines.append(
            f"| {r.name} | {r.expected:.4g} | {r.measured:.4g} | "
            f"{r.residual:.4g} | {r.tolerance:.4g} | {r.rays} | {status} |"
        )
    lines.append("")
    Path(path).write_text("\n".join(lines), encoding="utf-8")
```

```python
"""CLI entry: `python -m backlight_sim.golden [--report] [--out DIR] [--rays N] [--cases LIST]`.

Runs the case registry from backlight_sim.golden.cases::ALL_CASES and emits a
per-case PASS/FAIL summary. With --report, also writes an HTML + markdown
report via backlight_sim.golden.report.

Stdlib argparse only (no typer/click — no new deps per RESEARCH.md §Standard Stack).
Mirrors build_exe.py argparse pattern (the only other CLI in the repo).
"""
from __future__ import annotations

import argparse
import sys
from datetime import datetime
from pathlib import Path

from backlight_sim.golden.cases import ALL_CASES, run_case
from backlight_sim.golden.report import write_html_report, write_markdown_report


def _default_out_dir() -> Path:
    return Path("golden_reports") / datetime.now().strftime("%Y%m%d_%H%M%S")


def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(
        prog="python -m backlight_sim.golden",
        description="Run the golden-reference validation suite and emit a report.",
    )
    parser.add_argument(
        "--report", action="store_true",
        help="Write HTML + markdown summary (default: print to stdout only)",
    )
    parser.add_argument(
        "--out", type=Path, default=None,
        help="Output directory (default: ./golden_reports/<timestamp>)",
    )
    parser.add_argument(
        "--rays", type=int, default=None,
        help="Override per-case ray count (smaller = faster smoke run)",
    )
    parser.add_argument(
        "--cases", type=str, default=None,
        help="Comma-separated case names (default: all registered cases)",
    )
    parser.add_argument(
        "-v", "--verbose", action="store_true",
        help="Print per-case progress lines",
    )
    args = parser.parse_args(argv)

    if not ALL_CASES:
        print("[golden] ERROR: no cases registered in ALL_CASES — "
              "is backlight_sim.golden.cases correctly populated?",
              file=sys.stderr)
        return 2

    if args.cases is None:
        cases = list(ALL_CASES)
    else:
        wanted = {c.strip() for c in args.cases.split(",") if c.strip()}
        cases = [c for c in ALL_CASES if c.name in wanted]
        if not cases:
            print(f"[golden] ERROR: no cases matched {sorted(wanted)}", file=sys.stderr)
            return 2

    results = []
    for case in cases:
        gr = run_case(case, rays_override=args.rays, verbose=args.verbose)
        results.append(gr)
        if not args.verbose:
            status = "PASS" if gr.passed else "FAIL"
            print(f"  [{status}] {gr.name}: residual={gr.residual:.4g} "
                  f"tol={gr.tolerance:.4g} rays={gr.rays}")

        write_markdown_report(results, md_path)
```
