    ```python
    def _per_batch_source_flux(result, det) -> np.ndarray | None:
        """Return (n_batches,) per-batch source flux using rays_per_batch for unbiased scaling.

        When rays_per_batch is None OR rays_total == 0: returns None (caller falls back).
        Formula: source_flux_per_batch[k] = total_emitted_flux * rays_per_batch[k] / rays_total.
        This is UNBIASED even when rays_per_source % K != 0 (checker I5).
        """
        if det.rays_per_batch is None or result.total_emitted_flux <= 0:
            return None
        rpb = np.asarray(det.rays_per_batch, dtype=float)
        rays_total = rpb.sum()
        if rays_total <= 0:
            return None
        return result.total_emitted_flux * (rpb / rays_total)
    ```

    Then inside HeatmapPanel:
    ```python
    def _compute_all_kpi_batches(self, result: SimulationResult) -> dict[str, np.ndarray]:
        """Compute per-batch KPI arrays for every scalar KPI. Returns {} when UQ off."""
        out: dict[str, np.ndarray] = {}
        if not result.detectors:
            return out
        det = next(iter(result.detectors.values()))
        gb = det.grid_batches
        if gb is None or det.n_batches < 4:
            return out

        out["avg"]  = kpi_batches(gb, lambda g: float(g.mean()))
        out["peak"] = kpi_batches(gb, lambda g: float(g.max()))
        out["min"]  = kpi_batches(gb, lambda g: float(g.min()))
        out["cv"]   = kpi_batches(gb, lambda g: (float(g.std()) / float(g.mean()) * 100.0) if g.mean() > 0 else 0.0)
        out["hot"]  = kpi_batches(gb, lambda g: (float(g.max()) / float(g.mean())) if g.mean() > 0 else 0.0)
        out["ecr"]  = kpi_batches(gb, edge_center_ratio)
        out["corner"] = kpi_batches(gb, corner_ratio)
        # Uniformity rows:
        for label, frac in _FRACTIONS:
            out[f"uni_{label}_min_avg"] = kpi_batches(gb, lambda g, f=frac: uniformity_in_center(g, f)[0])
            out[f"uni_{label}_min_max"] = kpi_batches(gb, lambda g, f=frac: uniformity_in_center(g, f)[1])
        # Efficiency (per-batch) — use rays_per_batch for unbiased scaling (checker I5):
        from backlight_sim.core.kpi import _per_batch_source_flux
        per_batch_src = _per_batch_source_flux(result, det)
        if per_batch_src is not None and det.flux_batches is not None:
            # Avoid divide-by-zero on per-batch basis.
            eff = np.zeros_like(per_batch_src)
            nonzero = per_batch_src > 0
            eff[nonzero] = det.flux_batches[nonzero] / per_batch_src[nonzero] * 100.0
            out["eff"] = eff
        return out
    ```

    **Important (checker I5):** The OLD formula `per_batch_src = total_emitted_flux / n_batches` assumes equal per-batch ray counts. When Wave 2 distributes remainder to the first `rays_total % K` batches, actual per-batch source flux differs by ±1 ray worth of flux. For small K (e.g., K=10) and uneven rays_per_source, this introduces ~0.1–1% systematic bias in per-batch efficiency values — enough to skew the CI half-width estimate on tight simulations. The `_per_batch_source_flux` helper uses `rays_per_batch[k] / sum(rays_per_batch) * total_emitted_flux` which is EXACTLY proportional to the actual rays emitted per batch, eliminating the bias regardless of remainder distribution.

    ```python
    def compute_all_kpi_cis(result: SimulationResult, conf_level: float = 0.95) -> dict[str, "CIEstimate | None"]:
        """Return dict of CIEstimate for every scalar KPI. None values when UQ is off.

        Uses DetectorResult.rays_per_batch for unbiased per-batch efficiency scaling
        (checker I5) — NOT naive total_emitted_flux / n_batches.
        """
        from backlight_sim.core.uq import batch_mean_ci, kpi_batches, CIEstimate
        keys_default = ("avg", "peak", "min", "cv", "hot", "ecr", "corner",
                        "uni_1_4_min_avg", "uni_1_6_min_avg", "uni_1_10_min_avg",
                        "efficiency_pct")
        if not result.detectors:
            return {k: None for k in keys_default}
        det = next(iter(result.detectors.values()))
        gb = det.grid_batches
        if gb is None or det.n_batches < 4:
            return {k: None for k in keys_default}

        out: dict[str, CIEstimate | None] = {}
        out["avg"]  = batch_mean_ci(kpi_batches(gb, lambda g: float(g.mean())), conf_level)
        out["peak"] = batch_mean_ci(kpi_batches(gb, lambda g: float(g.max())), conf_level)
        out["min"]  = batch_mean_ci(kpi_batches(gb, lambda g: float(g.min())), conf_level)
        out["cv"]   = batch_mean_ci(kpi_batches(gb, lambda g: (float(g.std()) / float(g.mean()) * 100.0) if g.mean() > 0 else 0.0), conf_level)
        out["hot"]  = batch_mean_ci(kpi_batches(gb, lambda g: (float(g.max()) / float(g.mean())) if g.mean() > 0 else 0.0), conf_level)
        out["ecr"]  = batch_mean_ci(kpi_batches(gb, edge_center_ratio), conf_level)
        out["corner"] = batch_mean_ci(kpi_batches(gb, corner_ratio), conf_level)
        # Uniformity at common fractions:
        for label, frac in (("1_4", 0.25), ("1_6", 1/6), ("1_10", 0.1)):
            vals = kpi_batches(gb, lambda g, f=frac: uniformity_in_center(g, f)[0])
            out[f"uni_{label}_min_avg"] = batch_mean_ci(vals, conf_level)
        # Efficiency — rays_per_batch-aware (checker I5):
        per_batch_src = _per_batch_source_flux(result, det)
        if per_batch_src is not None and det.flux_batches is not None:
            with np.errstate(divide="ignore", invalid="ignore"):
                eff_batches = np.where(per_batch_src > 0,
                                       det.flux_batches / per_batch_src * 100.0, 0.0)
            out["efficiency_pct"] = batch_mean_ci(eff_batches, conf_level)
        else:
            out["efficiency_pct"] = None
        return out
    ```
