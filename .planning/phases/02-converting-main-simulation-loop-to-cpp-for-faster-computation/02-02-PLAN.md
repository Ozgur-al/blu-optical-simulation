#include "intersect.hpp"
#include <cmath>
#include <algorithm>

// ---------------------------------------------------------------------------
// Plane (Rectangle / DetectorSurface / SolidBox face)
// Ported from accel.py::intersect_plane_jit
// ---------------------------------------------------------------------------
double intersect_plane(
    double ox, double oy, double oz,
    double dx, double dy, double dz,
    double nx, double ny, double nz,
    double cx, double cy, double cz,
    double ux, double uy, double uz,
    double vx, double vy, double vz,
    double half_w, double half_h, double eps)
{
    double denom = dx*nx + dy*ny + dz*nz;
    if (std::abs(denom) <= 1e-12) return INF;
    double d_plane = nx*cx + ny*cy + nz*cz;
    double t = (d_plane - (ox*nx + oy*ny + oz*nz)) / denom;
    if (t <= eps) return INF;
    double hx = ox + t*dx - cx;
    double hy = oy + t*dy - cy;
    double hz = oz + t*dz - cz;
    double u_coord = hx*ux + hy*uy + hz*uz;
    double v_coord = hx*vx + hy*vy + hz*vz;
    if (std::abs(u_coord) <= half_w && std::abs(v_coord) <= half_h) return t;
    return INF;
}

// ---------------------------------------------------------------------------
// Disc (CylinderCap) — plane hit + radial clip
// ---------------------------------------------------------------------------
double intersect_disc(
    double ox, double oy, double oz,
    double dx, double dy, double dz,
    double cx, double cy, double cz,
    double nx, double ny, double nz,
    double radius, double eps)
{
    double denom = dx*nx + dy*ny + dz*nz;
    if (std::abs(denom) <= 1e-12) return INF;
    double d_plane = nx*cx + ny*cy + nz*cz;
    double t = (d_plane - (ox*nx + oy*ny + oz*nz)) / denom;
    if (t <= eps) return INF;
    double hx = ox + t*dx - cx;
    double hy = oy + t*dy - cy;
    double hz = oz + t*dz - cz;
    double r2 = hx*hx + hy*hy + hz*hz;
    if (r2 <= radius*radius) return t;
    return INF;
}

// ---------------------------------------------------------------------------
// Cylinder side (CylinderSide) — quadratic formula, axis-bounded
// Reference: PBR Book 3ed, Cylinders
// ---------------------------------------------------------------------------
double intersect_cylinder_side(
    double ox, double oy, double oz,
    double dx, double dy, double dz,
    double cx, double cy, double cz,
    double ax, double ay, double az,
    double radius, double half_length, double eps)
{
    // Project ray onto plane perpendicular to cylinder axis
    double ocx = ox - cx, ocy = oy - cy, ocz = oz - cz;
    double d_dot_a = dx*ax + dy*ay + dz*az;
    double oc_dot_a = ocx*ax + ocy*ay + ocz*az;
    // d_perp = d - (d.a)*a;  oc_perp = oc - (oc.a)*a
    double dpx = dx - d_dot_a*ax, dpy = dy - d_dot_a*ay, dpz = dz - d_dot_a*az;
    double opx = ocx - oc_dot_a*ax, opy = ocy - oc_dot_a*ay, opz = ocz - oc_dot_a*az;
    double a = dpx*dpx + dpy*dpy + dpz*dpz;
    if (a < 1e-14) return INF;  // ray parallel to axis
    double b = 2.0*(dpx*opx + dpy*opy + dpz*opz);
    double c = opx*opx + opy*opy + opz*opz - radius*radius;
    double disc = b*b - 4.0*a*c;
    if (disc < 0.0) return INF;
    double sqrt_disc = std::sqrt(disc);
    double inv2a = 1.0 / (2.0*a);
    double t1 = (-b - sqrt_disc)*inv2a;
    double t2 = (-b + sqrt_disc)*inv2a;
    // Pick smallest positive t that satisfies axis bounds
    for (double t_cand : {t1, t2}) {
        if (t_cand <= eps) continue;
        // Check axis bound: |proj_along_axis| <= half_length
        double hx = ox + t_cand*dx, hy = oy + t_cand*dy, hz = oz + t_cand*dz;
        double proj = (hx-cx)*ax + (hy-cy)*ay + (hz-cz)*az;
        if (std::abs(proj) <= half_length) return t_cand;
    }
    return INF;
}

// ---------------------------------------------------------------------------
// Sphere (SphereDetector)
// Ported from accel.py::intersect_sphere_jit
// ---------------------------------------------------------------------------
double intersect_sphere(
    double ox, double oy, double oz,
    double dx, double dy, double dz,
    double sx, double sy, double sz,
    double radius, double eps)
{
    double lx = ox - sx, ly = oy - sy, lz = oz - sz;
    double a = dx*dx + dy*dy + dz*dz;
    double b = 2.0*(dx*lx + dy*ly + dz*lz);
    double c = lx*lx + ly*ly + lz*lz - radius*radius;
    double disc = b*b - 4.0*a*c;
    if (disc < 0.0) return INF;
    double sqrt_disc = std::sqrt(disc);
    double inv2a = 1.0/(2.0*a);
    double t1 = (-b - sqrt_disc)*inv2a;
    double t2 = (-b + sqrt_disc)*inv2a;
    if (t1 > eps) return t1;
    if (t2 > eps) return t2;
    return INF;
}

// ---------------------------------------------------------------------------
// Prism cap (polygon containment via half-plane test)
// PrismCap.vertices_2d and edge_normals_2d are precomputed in solid_body.py
// ---------------------------------------------------------------------------
double intersect_prism_cap(
    double ox, double oy, double oz,
    double dx, double dy, double dz,
    const ScenePrismCap& cap, double eps)
{
    // First: plane intersection using cap normal and center
    double denom = dx*cap.nx + dy*cap.ny + dz*cap.nz;
    if (std::abs(denom) <= 1e-12) return INF;
    double d_plane = cap.nx*cap.cx + cap.ny*cap.cy + cap.nz*cap.cz;
    double t = (d_plane - (ox*cap.nx + oy*cap.ny + oz*cap.nz)) / denom;
    if (t <= eps) return INF;
    // Hit point in world coords
    double hx = ox + t*dx - cap.cx;
    double hy = oy + t*dy - cap.cy;
    double hz = oz + t*dz - cap.cz;
    // Project into local u/v 2D frame
    double u_c = hx*cap.ux + hy*cap.uy + hz*cap.uz;
    double v_c = hx*cap.vx + hy*cap.vy + hz*cap.vz;
    // Polygon containment: half-plane test against each edge
    for (int e = 0; e < cap.n_sides; ++e) {
        double vx2d = cap.vertices_2d[e][0];
        double vy2d = cap.vertices_2d[e][1];
        double enx = cap.edge_normals_2d[e][0];
        double eny = cap.edge_normals_2d[e][1];
        // Point relative to vertex, projected onto outward edge normal
        double dot = (u_c - vx2d)*enx + (v_c - vy2d)*eny;
        if (dot > 0.0) return INF;  // outside this edge
    }
    return t;
}

// ---------------------------------------------------------------------------
// AABB slab test (for BVH broad-phase filtering)
// ---------------------------------------------------------------------------
bool intersect_aabb(
    double ox, double oy, double oz,
    double dx, double dy, double dz,
    double xmin, double xmax,
    double ymin, double ymax,
    double zmin, double zmax)
{
    auto safe_inv = [](double d) {
        return std::abs(d) > 1e-15 ? 1.0/d : (d >= 0 ? 1e15 : -1e15);
    };
    double inv_dx = safe_inv(dx), inv_dy = safe_inv(dy), inv_dz = safe_inv(dz);
    double tx0 = (xmin - ox)*inv_dx, tx1 = (xmax - ox)*inv_dx;
    double ty0 = (ymin - oy)*inv_dy, ty1 = (ymax - oy)*inv_dy;
    double tz0 = (zmin - oz)*inv_dz, tz1 = (zmax - oz)*inv_dz;
    double tenter = std::max({std::min(tx0,tx1), std::min(ty0,ty1), std::min(tz0,tz1)});
    double texit  = std::min({std::max(tx0,tx1), std::max(ty0,ty1), std::max(tz0,tz1)});
    return texit >= tenter && texit > EPSILON;
}

#include "sampling.hpp"
#include <cmath>
#include <algorithm>
#include <numeric>

static const double PI = 3.14159265358979323846;

void build_basis(double wx, double wy, double wz,
                 double& ux, double& uy, double& uz,
                 double& vx, double& vy, double& vz)
{
    // Gram-Schmidt orthonormalization
    double ref_x = 1.0, ref_y = 0.0, ref_z = 0.0;
    double dot_wr = wx*ref_x + wy*ref_y + wz*ref_z;
    if (std::abs(dot_wr) > 0.9) { ref_x = 0.0; ref_y = 1.0; ref_z = 0.0; }
    // u = cross(w, ref); normalize
    ux = wy*ref_z - wz*ref_y;
    uy = wz*ref_x - wx*ref_z;
    uz = wx*ref_y - wy*ref_x;
    double norm_u = std::sqrt(ux*ux + uy*uy + uz*uz);
    ux /= norm_u; uy /= norm_u; uz /= norm_u;
    // v = cross(w, u)
    vx = wy*uz - wz*uy;
    vy = wz*ux - wx*uz;
    vz = wx*uy - wy*ux;
}

void sample_isotropic(int n, std::mt19937_64& rng, RayBatch& batch) {
    std::uniform_real_distribution<double> uni_z(-1.0, 1.0);
    std::uniform_real_distribution<double> uni_phi(0.0, 2.0*PI);
    for (int i = 0; i < n; ++i) {
        double z = uni_z(rng);
        double phi = uni_phi(rng);
        double r = std::sqrt(std::max(0.0, 1.0 - z*z));
        batch.dx[i] = r * std::cos(phi);
        batch.dy[i] = r * std::sin(phi);
        batch.dz[i] = z;
    }
}

void sample_lambertian(int n, double nx, double ny, double nz,
                       std::mt19937_64& rng, RayBatch& batch)
{
    double ux, uy, uz, vx, vy, vz;
    build_basis(nx, ny, nz, ux, uy, uz, vx, vy, vz);
    std::uniform_real_distribution<double> uni(0.0, 1.0);
    for (int i = 0; i < n; ++i) {
        double r = std::sqrt(uni(rng));
        double phi = uni(rng) * 2.0*PI;
        double xl = r * std::cos(phi);
        double yl = r * std::sin(phi);
        double zl = std::sqrt(std::max(0.0, 1.0 - xl*xl - yl*yl));
        batch.dx[i] = xl*ux + yl*vx + zl*nx;
        batch.dy[i] = xl*uy + yl*vy + zl*ny;
        batch.dz[i] = xl*uz + yl*vz + zl*nz;
    }
}

void sample_angular_distribution(int n,
                                  double dir_x, double dir_y, double dir_z,
                                  const std::vector<double>& theta_deg,
                                  const std::vector<double>& intensity,
                                  std::mt19937_64& rng, RayBatch& batch)
{
    if (theta_deg.size() < 2 || intensity.size() != theta_deg.size()) {
        sample_lambertian(n, dir_x, dir_y, dir_z, rng, batch);
        return;
    }
    // Build 2048-point grid and CDF (matching sampling.py exactly)
    const int NGRID = 2048;
    double theta_min = theta_deg.front() * PI / 180.0;
    double theta_max = theta_deg.back()  * PI / 180.0;
    std::vector<double> grid(NGRID), interp_i(NGRID);
    for (int k = 0; k < NGRID; ++k)
        grid[k] = theta_min + (theta_max - theta_min) * k / (NGRID - 1);

    // Linear interpolation of intensity onto grid
    auto interp = [&](double x) -> double {
        double x_deg = x * 180.0 / PI;
        // find bracket
        if (theta_deg.size() == 0) return 0.0;
        if (x_deg <= theta_deg.front()) return intensity.front();
        if (x_deg >= theta_deg.back())  return intensity.back();
        auto it = std::lower_bound(theta_deg.begin(), theta_deg.end(), x_deg);
        int hi = (int)(it - theta_deg.begin());
        int lo = hi - 1;
        double frac = (x_deg - theta_deg[lo]) / (theta_deg[hi] - theta_deg[lo]);
        return intensity[lo] + frac*(intensity[hi] - intensity[lo]);
    };

    std::vector<double> weights(NGRID), cdf(NGRID);
    for (int k = 0; k < NGRID; ++k)
        weights[k] = std::max(0.0, interp(grid[k])) * std::sin(grid[k]);
    double total = 0.0;
    for (int k = 0; k < NGRID; ++k) { total += weights[k]; cdf[k] = total; }
    if (total <= 0.0) { sample_lambertian(n, dir_x, dir_y, dir_z, rng, batch); return; }
    for (auto& c : cdf) c /= total;

    double ux, uy, uz, vx, vy, vz;
    build_basis(dir_x, dir_y, dir_z, ux, uy, uz, vx, vy, vz);
    std::uniform_real_distribution<double> uni(0.0, 1.0);
    for (int i = 0; i < n; ++i) {
        double u_val = uni(rng);
        // CDF inversion via lower_bound (O(log N))
        auto it = std::lower_bound(cdf.begin(), cdf.end(), u_val);
        int idx = std::min((int)(it - cdf.begin()), NGRID-1);
        double sample_theta = grid[idx];
        double phi = uni(rng) * 2.0*PI;
        double sin_t = std::sin(sample_theta);
        double cos_t = std::cos(sample_theta);
        double xl = sin_t * std::cos(phi);
        double yl = sin_t * std::sin(phi);
        double zl = cos_t;
        batch.dx[i] = xl*ux + yl*vx + zl*dir_x;
        batch.dy[i] = xl*uy + yl*vy + zl*dir_y;
        batch.dz[i] = xl*uz + yl*vz + zl*dir_z;
    }
}

void sample_lambertian_single(double nx, double ny, double nz,
                               std::mt19937_64& rng,
                               double& out_x, double& out_y, double& out_z)
{
    double ux, uy, uz, vx, vy, vz;
    build_basis(nx, ny, nz, ux, uy, uz, vx, vy, vz);
    std::uniform_real_distribution<double> uni(0.0, 1.0);
    double r = std::sqrt(uni(rng));
    double phi = uni(rng) * 2.0*PI;
    double xl = r * std::cos(phi);
    double yl = r * std::sin(phi);
    double zl = std::sqrt(std::max(0.0, 1.0 - xl*xl - yl*yl));
    out_x = xl*ux + yl*vx + zl*nx;
    out_y = xl*uy + yl*vy + zl*ny;
    out_z = xl*uz + yl*vz + zl*nz;
}

void reflect_specular(double dx, double dy, double dz,
                      double nx, double ny, double nz,
                      double& rdx, double& rdy, double& rdz)
{
    double dot = dx*nx + dy*ny + dz*nz;
    rdx = dx - 2.0*dot*nx;
    rdy = dy - 2.0*dot*ny;
    rdz = dz - 2.0*dot*nz;
    // Normalize
    double norm = std::sqrt(rdx*rdx + rdy*rdy + rdz*rdz);
    if (norm > 1e-12) { rdx /= norm; rdy /= norm; rdz /= norm; }
}

void scatter_haze_single(double dx, double dy, double dz,
                          double half_angle_deg,
                          std::mt19937_64& rng,
                          double& out_x, double& out_y, double& out_z)
{
    if (half_angle_deg <= 0.0) { out_x = dx; out_y = dy; out_z = dz; return; }
    double half_rad = half_angle_deg * 3.14159265358979 / 180.0;
    std::uniform_real_distribution<double> uni(0.0, 1.0);
    double cos_max = std::cos(half_rad);
    // Uniform cone sampling
    double cos_t = cos_max + (1.0 - cos_max)*uni(rng);
    double sin_t = std::sqrt(std::max(0.0, 1.0 - cos_t*cos_t));
    double phi = uni(rng) * 2.0 * 3.14159265358979;
    double ux, uy, uz, vx, vy, vz;
    build_basis(dx, dy, dz, ux, uy, uz, vx, vy, vz);
    out_x = sin_t*std::cos(phi)*ux + sin_t*std::sin(phi)*vx + cos_t*dx;
    out_y = sin_t*std::cos(phi)*uy + sin_t*std::sin(phi)*vy + cos_t*dy;
    out_z = sin_t*std::cos(phi)*uz + sin_t*std::sin(phi)*vz + cos_t*dz;
    double norm = std::sqrt(out_x*out_x + out_y*out_y + out_z*out_z);
    if (norm > 1e-12) { out_x /= norm; out_y /= norm; out_z /= norm; }
}

#include "material.hpp"
#include "sampling.hpp"
#include <cmath>

// Fresnel reflectance — ported from tracer.py::_fresnel_unpolarized
double fresnel_unpolarized(double cos_theta_i, double n1, double n2) {
    double cos_i = std::max(0.0, std::min(1.0, cos_theta_i));
    double ratio = n1 / n2;
    double sin_t_sq = ratio * ratio * (1.0 - cos_i*cos_i);
    if (sin_t_sq >= 1.0) return 1.0;  // TIR
    double cos_t = std::sqrt(std::max(0.0, 1.0 - sin_t_sq));
    double denom_s = std::max(1e-12, n1*cos_i + n2*cos_t);
    double denom_p = std::max(1e-12, n2*cos_i + n1*cos_t);
    double Rs = (n1*cos_i - n2*cos_t) / denom_s;
    double Rp = (n2*cos_i - n1*cos_t) / denom_p;
    return 0.5*(Rs*Rs + Rp*Rp);
}

// Snell refraction — ported from tracer.py::_refract_snell
bool refract_snell(
    double dx, double dy, double dz,
    double on_x, double on_y, double on_z,   // normal pointing into new medium
    double n1, double n2,
    double& rdx, double& rdy, double& rdz)
{
    double eta = n1 / n2;
    // cos_i = dot(d, on_into)  — convention from tracer.py line 148-153
    double cos_i = std::max(0.0, std::min(1.0, dx*on_x + dy*on_y + dz*on_z));
    double sin_t_sq = eta*eta*(1.0 - cos_i*cos_i);
    if (sin_t_sq >= 1.0) return false;  // TIR — caller must handle as reflection
    double cos_t = std::sqrt(std::max(0.0, 1.0 - sin_t_sq));
    // n_hat = -on_into (points toward incoming ray, into old medium)
    double nx_h = -on_x, ny_h = -on_y, nz_h = -on_z;
    rdx = eta*dx + (eta*cos_i - cos_t)*nx_h;
    rdy = eta*dy + (eta*cos_i - cos_t)*ny_h;
    rdz = eta*dz + (eta*cos_i - cos_t)*nz_h;
    double norm = std::sqrt(rdx*rdx + rdy*rdy + rdz*rdz);
    if (norm > 1e-12) { rdx /= norm; rdy /= norm; rdz /= norm; }
    return true;
}

// Material dispatch for regular surfaces (not solid body Fresnel)
// Matches tracer.py _bounce_surfaces logic for surface type 0
void apply_material(
    int i, RayBatch& batch,
    const SceneMaterial& mat,
    double hit_x, double hit_y, double hit_z,
    double face_nx, double face_ny, double face_nz,
    double geom_eps, std::mt19937_64& rng)
{
    if (mat.surface_type == "absorber") {
        batch.alive[i] = false;
        return;
    }
    // Orient normal toward incoming ray
    double dot_dn = batch.dx[i]*face_nx + batch.dy[i]*face_ny + batch.dz[i]*face_nz;
    double on_x = (dot_dn > 0) ? -face_nx : face_nx;
    double on_y = (dot_dn > 0) ? -face_ny : face_ny;
    double on_z = (dot_dn > 0) ? -face_nz : face_nz;

    if (mat.surface_type == "reflector") {
        batch.weights[i] *= mat.reflectance;
        double rdx, rdy, rdz;
        if (mat.is_diffuse) {
            sample_lambertian_single(on_x, on_y, on_z, rng, rdx, rdy, rdz);
        } else {
            reflect_specular(batch.dx[i], batch.dy[i], batch.dz[i],
                             on_x, on_y, on_z, rdx, rdy, rdz);
            if (mat.haze > 0.0) {
                scatter_haze_single(rdx, rdy, rdz, mat.haze, rng, rdx, rdy, rdz);
            }
        }
        batch.ox[i] = hit_x + on_x*geom_eps;
        batch.oy[i] = hit_y + on_y*geom_eps;
        batch.oz[i] = hit_z + on_z*geom_eps;
        batch.dx[i] = rdx; batch.dy[i] = rdy; batch.dz[i] = rdz;
    } else if (mat.surface_type == "diffuser") {
        std::uniform_real_distribution<double> uni(0.0, 1.0);
        if (uni(rng) < mat.transmittance) {
            // Transmit: Lambertian in through-direction
            double through_x = -on_x, through_y = -on_y, through_z = -on_z;
            double tdx, tdy, tdz;
            sample_lambertian_single(through_x, through_y, through_z, rng, tdx, tdy, tdz);
            batch.ox[i] = hit_x + through_x*geom_eps;
            batch.oy[i] = hit_y + through_y*geom_eps;
            batch.oz[i] = hit_z + through_z*geom_eps;
            batch.dx[i] = tdx; batch.dy[i] = tdy; batch.dz[i] = tdz;
        } else {
            // Reflect: Lambertian, weight * reflectance
            batch.weights[i] *= mat.reflectance;
            double rdx, rdy, rdz;
            sample_lambertian_single(on_x, on_y, on_z, rng, rdx, rdy, rdz);
            batch.ox[i] = hit_x + on_x*geom_eps;
            batch.oy[i] = hit_y + on_y*geom_eps;
            batch.oz[i] = hit_z + on_z*geom_eps;
            batch.dx[i] = rdx; batch.dy[i] = rdy; batch.dz[i] = rdz;
        }
    }
}

static constexpr int BVH_THRESHOLD = 9999;  // disabled for Wave 1

struct GridAccum {
    std::vector<double> grid;  // (ny * nx) flat, row-major
    int nx, ny;
    int hits;
    double flux;
};

struct SphGridAccum {
    std::vector<double> grid;  // (n_theta * n_phi) flat
    int n_phi, n_theta;
    int hits;
    double flux;
};

// Grid accumulation (mirrors accumulate_grid_jit from accel.py)
static void accumulate_grid(
    GridAccum& accum,
    double hit_x, double hit_y, double hit_z,
    double cx, double cy, double cz,
    double ux, double uy, double uz,
    double vx, double vy, double vz,
    double half_w, double half_h,
    double weight)
{
    double lx = hit_x - cx, ly = hit_y - cy, lz = hit_z - cz;
    double u_c = lx*ux + ly*uy + lz*uz;
    double v_c = lx*vx + ly*vy + lz*vz;
    int ix = (int)std::floor((u_c / (2.0*half_w) + 0.5) * accum.nx);
    int iy = (int)std::floor((v_c / (2.0*half_h) + 0.5) * accum.ny);
    ix = std::max(0, std::min(accum.nx - 1, ix));
    iy = std::max(0, std::min(accum.ny - 1, iy));
    accum.grid[iy * accum.nx + ix] += weight;
    accum.hits++;
    accum.flux += weight;
}

// Sphere detector accumulation (near-field mode — accumulate_sphere_jit pattern)
static void accumulate_sphere(SphGridAccum& accum,
    double hx, double hy, double hz,
    double cx, double cy, double cz,
    double radius, double weight)
{
    double dx = hx-cx, dy = hy-cy, dz = hz-cz;
    double norm = std::sqrt(dx*dx + dy*dy + dz*dz);
    if (norm < 1e-12) return;
    double phi = std::atan2(dy, dx);  // [-pi, pi]
    if (phi < 0) phi += 2.0*3.14159265358979;
    double theta = std::acos(std::max(-1.0, std::min(1.0, dz/norm)));  // [0, pi]
    int i_phi  = (int)std::floor(phi   / (2.0*3.14159265358979) * accum.n_phi);
    int i_theta = (int)std::floor(theta / 3.14159265358979      * accum.n_theta);
    i_phi  = std::max(0, std::min(accum.n_phi-1,  i_phi));
    i_theta = std::max(0, std::min(accum.n_theta-1, i_theta));
    accum.grid[i_theta * accum.n_phi + i_phi] += weight;
    accum.hits++;
    accum.flux += weight;
}

py::dict run_bounce_loop(
    const SceneSource& source,
    const std::vector<ScenePlane>& surfaces,
    const std::vector<ScenePlane>& detector_planes,
    const std::vector<SceneSphereDetector>& sphere_dets,
    const std::vector<SceneCylinderCap>& cyl_caps,
    const std::vector<SceneCylinderSide>& cyl_sides,
    const std::vector<ScenePrismCap>& prism_caps,
    const std::vector<ScenePlane>& prism_sides,
    const std::vector<ScenePlane>& solid_faces,
    const std::unordered_map<std::string, SceneMaterial>& materials,
    const std::unordered_map<std::string, SceneMaterial>& optical_props,
    const std::unordered_map<std::string, AngularProfile>& ang_dists,
    const std::vector<std::pair<int,int>>& det_resolutions,  // (nx, ny) per detector
    int n_rays, int max_bounces, double energy_threshold,
    uint64_t seed)
{
    std::mt19937_64 rng(seed);

    // Initialize detector accumulators
    std::vector<GridAccum> det_accums(detector_planes.size());
    for (int d = 0; d < (int)detector_planes.size(); ++d) {
        det_accums[d].nx = det_resolutions[d].first;
        det_accums[d].ny = det_resolutions[d].second;
        det_accums[d].grid.assign(det_accums[d].nx * det_accums[d].ny, 0.0);
        det_accums[d].hits = 0;
        det_accums[d].flux = 0.0;
    }

    // Initialize sphere detector accumulators
    std::vector<SphGridAccum> sph_accums(sphere_dets.size());
    for (int s = 0; s < (int)sphere_dets.size(); ++s) {
        sph_accums[s].n_phi  = sphere_dets[s].n_phi;
        sph_accums[s].n_theta = sphere_dets[s].n_theta;
        sph_accums[s].grid.assign(sph_accums[s].n_phi * sph_accums[s].n_theta, 0.0);
        sph_accums[s].hits = 0;
        sph_accums[s].flux = 0.0;
    }

    double escaped_flux = 0.0;
    double weight_per_ray = source.effective_flux / (double)n_rays;

    // ---- Allocate RayBatch ----
    RayBatch batch;
    batch.n = n_rays;
    batch.ox.resize(n_rays); batch.oy.resize(n_rays); batch.oz.resize(n_rays);
    batch.dx.resize(n_rays); batch.dy.resize(n_rays); batch.dz.resize(n_rays);
    batch.weights.resize(n_rays, weight_per_ray);
    batch.alive.resize(n_rays, true);
    batch.current_n.resize(n_rays, 1.0);
    batch.n_stack.resize(n_rays);
    for (auto& s : batch.n_stack) s.fill(1.0);
    batch.n_depth.resize(n_rays, 0);

    // ---- Emit rays ----
    for (int i = 0; i < n_rays; ++i) {
        batch.ox[i] = source.px;
        batch.oy[i] = source.py;
        batch.oz[i] = source.pz;
    }
    if (source.distribution == "lambertian") {
        sample_lambertian(n_rays, source.dir_x, source.dir_y, source.dir_z, rng, batch);
    } else if (source.distribution == "isotropic") {
        sample_isotropic(n_rays, rng, batch);
    } else {
        auto it = ang_dists.find(source.distribution);
        if (it != ang_dists.end()) {
            sample_angular_distribution(n_rays, source.dir_x, source.dir_y, source.dir_z,
                                        it->second.theta_deg, it->second.intensity, rng, batch);
        } else {
            sample_lambertian(n_rays, source.dir_x, source.dir_y, source.dir_z, rng, batch);
        }
    }

    // ---- Bounce loop ----
    std::vector<int> active_idx;
    active_idx.reserve(n_rays);

    for (int bounce = 0; bounce < max_bounces; ++bounce) {
        active_idx.clear();
        for (int i = 0; i < n_rays; ++i)
            if (batch.alive[i]) active_idx.push_back(i);
        if (active_idx.empty()) break;

        int n_active = (int)active_idx.size();
        std::vector<double> best_t(n_active, INF);
        std::vector<int> best_type(n_active, -1);  // 0=surf, 1=det, 2=sphere, 3=solidface, 4=cylcap, 5=cylside, 6=prismcap, 7=prismside
        std::vector<int> best_obj(n_active, -1);

        // --- Test all surfaces (brute-force) ---
        for (int si = 0; si < (int)surfaces.size(); ++si) {
            const auto& surf = surfaces[si];
            for (int ri = 0; ri < n_active; ++ri) {
                int i = active_idx[ri];
                double t = intersect_plane(
                    batch.ox[i], batch.oy[i], batch.oz[i],
                    batch.dx[i], batch.dy[i], batch.dz[i],
                    surf.nx, surf.ny, surf.nz,
                    surf.cx, surf.cy, surf.cz,
                    surf.ux, surf.uy, surf.uz,
                    surf.vx, surf.vy, surf.vz,
                    surf.half_w, surf.half_h, EPSILON);
                if (t < best_t[ri]) { best_t[ri]=t; best_type[ri]=0; best_obj[ri]=si; }
            }
        }
        // --- Test solid body faces ---
        for (int sfi = 0; sfi < (int)solid_faces.size(); ++sfi) {
            const auto& sface = solid_faces[sfi];
            for (int ri = 0; ri < n_active; ++ri) {
                int i = active_idx[ri];
                double t = intersect_plane(
                    batch.ox[i], batch.oy[i], batch.oz[i],
                    batch.dx[i], batch.dy[i], batch.dz[i],
                    sface.nx, sface.ny, sface.nz,
                    sface.cx, sface.cy, sface.cz,
                    sface.ux, sface.uy, sface.uz,
                    sface.vx, sface.vy, sface.vz,
                    sface.half_w, sface.half_h, sface.geom_eps);
                if (t < best_t[ri]) { best_t[ri]=t; best_type[ri]=3; best_obj[ri]=sfi; }
            }
        }
        // --- Test detector planes ---
        for (int di = 0; di < (int)detector_planes.size(); ++di) {
            const auto& det = detector_planes[di];
            for (int ri = 0; ri < n_active; ++ri) {
                int i = active_idx[ri];
                double t = intersect_plane(
                    batch.ox[i], batch.oy[i], batch.oz[i],
                    batch.dx[i], batch.dy[i], batch.dz[i],
                    det.nx, det.ny, det.nz,
                    det.cx, det.cy, det.cz,
                    det.ux, det.uy, det.uz,
                    det.vx, det.vy, det.vz,
                    det.half_w, det.half_h, EPSILON);
                if (t < best_t[ri]) { best_t[ri]=t; best_type[ri]=1; best_obj[ri]=di; }
            }
        }
        // --- Test sphere detectors ---
        for (int sdi = 0; sdi < (int)sphere_dets.size(); ++sdi) {
            const auto& sd = sphere_dets[sdi];
            for (int ri = 0; ri < n_active; ++ri) {
                int i = active_idx[ri];
                double t = intersect_sphere(
                    batch.ox[i], batch.oy[i], batch.oz[i],
                    batch.dx[i], batch.dy[i], batch.dz[i],
                    sd.cx, sd.cy, sd.cz, sd.radius, EPSILON);
                if (t < best_t[ri]) { best_t[ri]=t; best_type[ri]=2; best_obj[ri]=sdi; }
            }
        }
        // --- Test cylinder caps ---
        for (int ci = 0; ci < (int)cyl_caps.size(); ++ci) {
            const auto& cap = cyl_caps[ci];
            for (int ri = 0; ri < n_active; ++ri) {
                int i = active_idx[ri];
                double t = intersect_disc(
                    batch.ox[i], batch.oy[i], batch.oz[i],
                    batch.dx[i], batch.dy[i], batch.dz[i],
                    cap.cx, cap.cy, cap.cz,
                    cap.nx, cap.ny, cap.nz,
                    cap.radius, cap.geom_eps);
                if (t < best_t[ri]) { best_t[ri]=t; best_type[ri]=4; best_obj[ri]=ci; }
            }
        }
        // --- Test cylinder sides ---
        for (int csi = 0; csi < (int)cyl_sides.size(); ++csi) {
            const auto& cs = cyl_sides[csi];
            for (int ri = 0; ri < n_active; ++ri) {
                int i = active_idx[ri];
                double t = intersect_cylinder_side(
                    batch.ox[i], batch.oy[i], batch.oz[i],
                    batch.dx[i], batch.dy[i], batch.dz[i],
                    cs.cx, cs.cy, cs.cz,
                    cs.ax, cs.ay, cs.az,
                    cs.radius, cs.half_length, cs.geom_eps);
                if (t < best_t[ri]) { best_t[ri]=t; best_type[ri]=5; best_obj[ri]=csi; }
            }
        }
        // --- Test prism caps ---
        for (int pci = 0; pci < (int)prism_caps.size(); ++pci) {
            const auto& pcap = prism_caps[pci];
            for (int ri = 0; ri < n_active; ++ri) {
                int i = active_idx[ri];
                double t = intersect_prism_cap(
                    batch.ox[i], batch.oy[i], batch.oz[i],
                    batch.dx[i], batch.dy[i], batch.dz[i],
                    pcap, pcap.geom_eps);
                if (t < best_t[ri]) { best_t[ri]=t; best_type[ri]=6; best_obj[ri]=pci; }
            }
        }
        // --- Test prism side faces ---
        for (int psi = 0; psi < (int)prism_sides.size(); ++psi) {
            const auto& pside = prism_sides[psi];
            for (int ri = 0; ri < n_active; ++ri) {
                int i = active_idx[ri];
                double t = intersect_plane(
                    batch.ox[i], batch.oy[i], batch.oz[i],
                    batch.dx[i], batch.dy[i], batch.dz[i],
                    pside.nx, pside.ny, pside.nz,
                    pside.cx, pside.cy, pside.cz,
                    pside.ux, pside.uy, pside.uz,
                    pside.vx, pside.vy, pside.vz,
                    pside.half_w, pside.half_h, pside.geom_eps);
                if (t < best_t[ri]) { best_t[ri]=t; best_type[ri]=7; best_obj[ri]=psi; }
            }
        }

        // --- Process hits ---
        for (int ri = 0; ri < n_active; ++ri) {
            int i = active_idx[ri];
            if (best_t[ri] == INF) {
                // Ray escaped
                escaped_flux += batch.weights[i];
                batch.alive[i] = false;
                continue;
            }
            double t = best_t[ri];
            double hit_x = batch.ox[i] + t*batch.dx[i];
            double hit_y = batch.oy[i] + t*batch.dy[i];
            double hit_z = batch.oz[i] + t*batch.dz[i];
            int type = best_type[ri];
            int obj  = best_obj[ri];

            if (type == 1) {
                // Detector hit — accumulate, advance ray past detector (pass-through)
                const auto& det = detector_planes[obj];
                accumulate_grid(det_accums[obj],
                    hit_x, hit_y, hit_z,
                    det.cx, det.cy, det.cz,
                    det.ux, det.uy, det.uz,
                    det.vx, det.vy, det.vz,
                    det.half_w, det.half_h,
                    batch.weights[i]);
                // Pass-through: advance origin past detector by EPSILON
                batch.ox[i] = hit_x + batch.dx[i]*EPSILON;
                batch.oy[i] = hit_y + batch.dy[i]*EPSILON;
                batch.oz[i] = hit_z + batch.dz[i]*EPSILON;
                continue;
            } else if (type == 2) {
                // Sphere detector hit — accumulate, pass-through
                const auto& sd = sphere_dets[obj];
                accumulate_sphere(sph_accums[obj],
                    hit_x, hit_y, hit_z,
                    sd.cx, sd.cy, sd.cz, sd.radius, batch.weights[i]);
                batch.ox[i] = hit_x + batch.dx[i]*EPSILON;
                batch.oy[i] = hit_y + batch.dy[i]*EPSILON;
                batch.oz[i] = hit_z + batch.dz[i]*EPSILON;
                continue;
            } else if (type == 0) {
                // Surface hit — material dispatch
                const auto& surf = surfaces[obj];
                // Resolve material (optical_properties_name takes priority)
                const SceneMaterial* mat_ptr = nullptr;
                if (!surf.optical_properties_name.empty()) {
                    auto it = optical_props.find(surf.optical_properties_name);
                    if (it != optical_props.end()) mat_ptr = &it->second;
                }
                if (!mat_ptr) {
                    auto it = materials.find(surf.material_name);
                    if (it != materials.end()) mat_ptr = &it->second;
                }
                if (!mat_ptr) {
                    batch.alive[i] = false;  // No material = absorb
                    continue;
                }
                apply_material(i, batch, *mat_ptr,
                    hit_x, hit_y, hit_z,
                    surf.nx, surf.ny, surf.nz,
                    EPSILON, rng);
            } else if (type == 3 || type == 4 || type == 5 || type == 6 || type == 7) {
                // Solid body face hit — Fresnel/TIR dispatch
                // Determine face normal and solid body refractive index
                double fnx, fny, fnz, box_n, geom_eps_sb;
                if (type == 3) {
                    fnx=solid_faces[obj].nx; fny=solid_faces[obj].ny; fnz=solid_faces[obj].nz;
                    box_n = solid_faces[obj].refractive_index;
                    geom_eps_sb = solid_faces[obj].geom_eps;
                } else if (type == 4) {
                    fnx=cyl_caps[obj].nx; fny=cyl_caps[obj].ny; fnz=cyl_caps[obj].nz;
                    box_n = cyl_caps[obj].refractive_index;
                    geom_eps_sb = cyl_caps[obj].geom_eps;
                } else if (type == 5) {
                    // CylinderSide: normal = perpendicular from axis to hit point
                    const auto& cs = cyl_sides[obj];
                    double oc_x=hit_x-cs.cx, oc_y=hit_y-cs.cy, oc_z=hit_z-cs.cz;
                    double proj = oc_x*cs.ax + oc_y*cs.ay + oc_z*cs.az;
                    double rn_x=oc_x-proj*cs.ax, rn_y=oc_y-proj*cs.ay, rn_z=oc_z-proj*cs.az;
                    double rn_len = std::sqrt(rn_x*rn_x + rn_y*rn_y + rn_z*rn_z);
                    fnx = (rn_len>1e-12) ? rn_x/rn_len : 1.0;
                    fny = (rn_len>1e-12) ? rn_y/rn_len : 0.0;
                    fnz = (rn_len>1e-12) ? rn_z/rn_len : 0.0;
                    box_n = cs.refractive_index;
                    geom_eps_sb = cs.geom_eps;
                } else if (type == 6) {
                    fnx=prism_caps[obj].nx; fny=prism_caps[obj].ny; fnz=prism_caps[obj].nz;
                    box_n = prism_caps[obj].refractive_index;
                    geom_eps_sb = prism_caps[obj].geom_eps;
                } else {
                    fnx=prism_sides[obj].nx; fny=prism_sides[obj].ny; fnz=prism_sides[obj].nz;
                    box_n = prism_sides[obj].refractive_index;
                    geom_eps_sb = prism_sides[obj].geom_eps;
                }

                // Entering vs exiting: dot(d, face_normal) < 0 → entering
                double dot_dn = batch.dx[i]*fnx + batch.dy[i]*fny + batch.dz[i]*fnz;
                bool entering = dot_dn < 0;
                // on_into: normal pointing into new medium
                double on_x = entering ? -fnx : fnx;
                double on_y = entering ? -fny : fny;
                double on_z = entering ? -fnz : fnz;
                double on_b_x = -on_x, on_b_y = -on_y, on_b_z = -on_z;  // back into old medium

                double n1 = batch.current_n[i];
                double exit_n = 1.0;
                if (batch.n_depth[i] > 0) {
                    exit_n = batch.n_stack[i][batch.n_depth[i] - 1];
                }
                double n2 = entering ? box_n : exit_n;

                double cos_i = std::max(0.0, std::min(1.0, batch.dx[i]*on_x + batch.dy[i]*on_y + batch.dz[i]*on_z));
                double R = fresnel_unpolarized(cos_i, n1, n2);

                std::uniform_real_distribution<double> uni(0.0, 1.0);
                bool reflects = (uni(rng) < R);

                if (!reflects) {
                    // Refract
                    double rdx, rdy, rdz;
                    bool ok = refract_snell(batch.dx[i], batch.dy[i], batch.dz[i],
                                            on_x, on_y, on_z, n1, n2, rdx, rdy, rdz);
                    if (!ok) reflects = true;  // TIR fallback
                    else {
                        batch.current_n[i] = n2;
                        // Update n_stack
                        if (entering && batch.n_depth[i] < N_STACK_MAX) {
                            batch.n_stack[i][batch.n_depth[i]] = n1;
                            batch.n_depth[i]++;
                        } else if (!entering && batch.n_depth[i] > 0) {
                            batch.n_depth[i]--;
                        }
                        batch.ox[i] = hit_x + on_x * geom_eps_sb;
                        batch.oy[i] = hit_y + on_y * geom_eps_sb;
                        batch.oz[i] = hit_z + on_z * geom_eps_sb;
                        batch.dx[i] = rdx; batch.dy[i] = rdy; batch.dz[i] = rdz;
                    }
                }
                if (reflects) {
                    // Specular reflection
                    double dot_vals = batch.dx[i]*on_b_x + batch.dy[i]*on_b_y + batch.dz[i]*on_b_z;
                    double rdx = batch.dx[i] - 2.0*dot_vals*on_b_x;
                    double rdy = batch.dy[i] - 2.0*dot_vals*on_b_y;
                    double rdz = batch.dz[i] - 2.0*dot_vals*on_b_z;
                    double rnorm = std::sqrt(rdx*rdx + rdy*rdy + rdz*rdz);
                    if (rnorm > 1e-12) { rdx /= rnorm; rdy /= rnorm; rdz /= rnorm; }
                    batch.ox[i] = hit_x + on_b_x * geom_eps_sb;
                    batch.oy[i] = hit_y + on_b_y * geom_eps_sb;
                    batch.oz[i] = hit_z + on_b_z * geom_eps_sb;
                    batch.dx[i] = rdx; batch.dy[i] = rdy; batch.dz[i] = rdz;
                }
            }
        }  // end hit processing

        // Energy threshold kill
        for (int i = 0; i < n_rays; ++i)
            if (batch.alive[i] && batch.weights[i] < energy_threshold)
                batch.alive[i] = false;
    }  // end bounce loop

    // ---- Build return dict ----
    py::dict grids;
    for (int d = 0; d < (int)detector_planes.size(); ++d) {
        const auto& accum = det_accums[d];
        npy_intp dims[2] = {(npy_intp)accum.ny, (npy_intp)accum.nx};
        py::array_t<double> arr(dims);
        auto buf = arr.request();
        std::copy(accum.grid.begin(), accum.grid.end(),
                  static_cast<double*>(buf.ptr));
        py::dict entry;
        entry["grid"] = arr;
        entry["hits"] = py::int_(accum.hits);
        entry["flux"] = py::float_(accum.flux);
        grids[detector_planes[d].name.c_str()] = entry;
    }

    py::dict sph_grids;
    for (int s = 0; s < (int)sphere_dets.size(); ++s) {
        const auto& accum = sph_accums[s];
        npy_intp dims[2] = {(npy_intp)accum.n_theta, (npy_intp)accum.n_phi};
        py::array_t<double> arr(dims);
        auto buf = arr.request();
        std::copy(accum.grid.begin(), accum.grid.end(),
                  static_cast<double*>(buf.ptr));
        py::dict entry;
        entry["grid"] = arr;
        entry["hits"] = py::int_(accum.hits);
        entry["flux"] = py::float_(accum.flux);
        sph_grids[sphere_dets[s].name.c_str()] = entry;
    }

    py::dict result;
    result["grids"] = grids;
    result["spectral_grids"] = py::dict();
    result["escaped"] = py::float_(escaped_flux);
    result["sb_stats"] = py::dict();
    result["sph_grids"] = sph_grids;
    return result;
}
