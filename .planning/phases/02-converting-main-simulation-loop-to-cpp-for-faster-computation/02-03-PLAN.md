```python
def _serialize_project(project) -> dict:
    """Serialize a Project dataclass to a plain dict for the C++ extension.

    The C++ trace_source() deserializes this dict — keep the key names and
    value types aligned with blu_tracer.cpp::trace_source deserialization.
    """
    def arr3(v):
        return list(float(x) for x in v)

    surfaces_list = []
    for s in project.surfaces:
        surfaces_list.append({
            "name": s.name,
            "center": arr3(s.center),
            "normal": arr3(s.normal),
            "u_axis": arr3(s.u_axis),
            "v_axis": arr3(s.v_axis),
            "size": [float(s.size[0]), float(s.size[1])],
            "material_name": str(s.material_name or ""),
            "optical_properties_name": str(getattr(s, "optical_properties_name", "") or ""),
        })

    detectors_list = []
    for d in project.detectors:
        detectors_list.append({
            "name": d.name,
            "center": arr3(d.center),
            "normal": arr3(d.normal),
            "u_axis": arr3(d.u_axis),
            "v_axis": arr3(d.v_axis),
            "size": [float(d.size[0]), float(d.size[1])],
            "resolution": list(d.resolution),
        })

    sphere_det_list = []
    for sd in getattr(project, "sphere_detectors", []):
        sphere_det_list.append({
            "name": sd.name,
            "center": arr3(sd.center),
            "radius": float(sd.radius),
            "resolution": list(sd.resolution),
            "mode": str(getattr(sd, "mode", "near_field")),
        })

    materials_dict = {}
    for name, mat in project.materials.items():
        materials_dict[name] = {
            "surface_type": str(mat.surface_type),
            "reflectance": float(mat.reflectance),
            "transmittance": float(mat.transmittance),
            "is_diffuse": bool(mat.is_diffuse),
            "haze": float(mat.haze),
            "refractive_index": float(getattr(mat, "refractive_index", 1.0)),
        }

    optical_props_dict = {}
    for name, op in (getattr(project, "optical_properties", {}) or {}).items():
        optical_props_dict[name] = {
            "surface_type": str(op.surface_type),
            "reflectance": float(op.reflectance),
            "transmittance": float(op.transmittance),
            "is_diffuse": bool(op.is_diffuse),
            "haze": float(op.haze),
            "refractive_index": float(getattr(op, "refractive_index", 1.0)),
        }

    ang_dist_dict = {}
    for name, profile in project.angular_distributions.items():
        ang_dist_dict[name] = {
            "theta_deg": [float(x) for x in profile.get("theta_deg", [])],
            "intensity": [float(x) for x in profile.get("intensity", [])],
        }

    sources_list = []
    for s in project.sources:
        sources_list.append({
            "name": s.name,
            "position": arr3(s.position),
            "direction": arr3(s.direction),
            "distribution": str(s.distribution),
            "effective_flux": float(s.effective_flux),
            "enabled": bool(s.enabled),
            "flux_tolerance": float(s.flux_tolerance),
            "spd": str(getattr(s, "spd", "white") or "white"),
        })

    settings = project.settings
    settings_dict = {
        "rays_per_source": int(settings.rays_per_source),
        "max_bounces": int(settings.max_bounces),
        "energy_threshold": float(settings.energy_threshold),
        "random_seed": int(settings.random_seed),
        "record_ray_paths": int(settings.record_ray_paths),
        "use_multiprocessing": bool(settings.use_multiprocessing),
        "adaptive_sampling": bool(getattr(settings, "adaptive_sampling", False)),
        "check_interval": int(getattr(settings, "check_interval", settings.rays_per_source)),
    }

    # Solid bodies: each SolidBox face is serialized as a flat surface-like dict
    solid_bodies_list = []
    for box in project.solid_bodies:
        mat = project.materials.get(box.material_name)
        box_n = float(mat.refractive_index) if mat else 1.0
        geom_eps = max(1e-6, min(box.dimensions) * 1e-4)
        faces_list = []
        for face in box.get_faces():
            face_id = face.name.split("::", 1)[1]
            faces_list.append({
                "name": face.name,
                "face_id": face_id,
                "center": arr3(face.center),
                "normal": arr3(face.normal),
                "u_axis": arr3(face.u_axis),
                "v_axis": arr3(face.v_axis),
                "size": [float(face.size[0]), float(face.size[1])],
                "optical_properties_name": str(getattr(face, "optical_properties_name", "") or ""),
            })
        solid_bodies_list.append({
            "name": box.name,
            "material_name": box.material_name,
            "refractive_index": box_n,
            "geom_eps": geom_eps,
            "faces": faces_list,
        })

    # Solid cylinders
    solid_cylinders_list = []
    for cyl in getattr(project, "solid_cylinders", []):
        mat = project.materials.get(cyl.material_name)
        cyl_n = float(mat.refractive_index) if mat else 1.0
        geom_eps = max(1e-6, min(cyl.radius, cyl.length / 2.0) * 1e-4)
        faces_list = []
        for face in cyl.get_faces():
            face_id = face.name.split("::", 1)[1]
            face_dict = {
                "name": face.name,
                "face_id": face_id,
                "face_type": "cap" if hasattr(face, "radius") and not hasattr(face, "axis") else (
                    "side" if hasattr(face, "axis") else "unknown"),
                "center": arr3(face.center),
                "optical_properties_name": str(getattr(face, "optical_properties_name", "") or ""),
            }
            if hasattr(face, "radius") and not hasattr(face, "axis"):
                # CylinderCap
                face_dict["face_type"] = "cap"
                face_dict["normal"] = arr3(face.normal)
                face_dict["radius"] = float(face.radius)
            else:
                # CylinderSide
                face_dict["face_type"] = "side"
                face_dict["axis"] = arr3(face.axis)
                face_dict["radius"] = float(face.radius)
                face_dict["length"] = float(face.length)
            faces_list.append(face_dict)
        solid_cylinders_list.append({
            "name": cyl.name,
            "material_name": cyl.material_name,
            "refractive_index": cyl_n,
            "geom_eps": geom_eps,
            "faces": faces_list,
        })

    # Solid prisms
    solid_prisms_list = []
    for prism in getattr(project, "solid_prisms", []):
        mat = project.materials.get(prism.material_name)
        prism_n = float(mat.refractive_index) if mat else 1.0
        geom_eps = max(1e-6, min(prism.circumscribed_radius, prism.length / 2.0) * 1e-4)
        faces_list = []
        for face in prism.get_faces():
            face_id = face.name.split("::", 1)[1]
            face_dict = {
                "name": face.name,
                "face_id": face_id,
                "center": arr3(face.center),
                "normal": arr3(face.normal),
                "optical_properties_name": str(getattr(face, "optical_properties_name", "") or ""),
            }
            if isinstance(face, PrismCap):
                face_dict["face_type"] = "prism_cap"
                face_dict["u_axis"] = arr3(face.u_axis)
                face_dict["v_axis"] = arr3(face.v_axis)
                face_dict["n_sides"] = int(face.n_sides)
                face_dict["vertices_2d"] = face.vertices_2d.tolist()
                face_dict["edge_normals_2d"] = face.edge_normals_2d.tolist()
            else:
                # Rectangle side face
                face_dict["face_type"] = "rect"
                face_dict["u_axis"] = arr3(face.u_axis)
                face_dict["v_axis"] = arr3(face.v_axis)
                face_dict["size"] = [float(face.size[0]), float(face.size[1])]
            faces_list.append(face_dict)
        solid_prisms_list.append({
            "name": prism.name,
            "material_name": prism.material_name,
            "refractive_index": prism_n,
            "geom_eps": geom_eps,
            "faces": faces_list,
        })

    return {
        "sources": sources_list,
        "surfaces": surfaces_list,
        "detectors": detectors_list,
        "sphere_detectors": sphere_det_list,
        "materials": materials_dict,
        "optical_properties": optical_props_dict,
        "angular_distributions": ang_dist_dict,
        "solid_bodies": solid_bodies_list,
        "solid_cylinders": solid_cylinders_list,
        "solid_prisms": solid_prisms_list,
        "settings": settings_dict,
    }
```

```python
def _cpp_trace_single_source(project, source_name: str, base_seed: int) -> dict:
    """Top-level function for multiprocessing: trace one source via C++ extension.

    Mirrors _trace_single_source but delegates to _blu_tracer.trace_source().
    Returns the same dict shape as _trace_single_source.
    """
    import hashlib
    seed_hash = int(hashlib.md5(f"{base_seed}_{source_name}".encode()).hexdigest()[:8], 16)
    project_dict = _serialize_project(project)
    return _blu_tracer.trace_source(project_dict, source_name, int(seed_hash))
```
